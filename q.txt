            function formatTime(sec) {
                if (!sec || isNaN(sec)) return '0:00';
                const m = Math.floor(sec / 60);
                const s = Math.floor(sec % 60);
                return m + ':' + (s<10?'0':'') + s;
            }

            function getNextIndex(current, mode, playedList) {
                if (mode === 'random') {
                    if (playedList.length >= songs.length) playedList.length = 0;
                    let available = songs.map((_,i)=>i).filter(i => !playedList.includes(i));
                    if (available.length === 0) { playedList.length = 0; available = songs.map((_,i)=>i); }
                    const rand = available[Math.floor(Math.random()*available.length)];
                    playedList.push(rand);
                    return rand;
                }
                if (mode === 'loop') return current;
                return (current + 1) % songs.length;
            }

            function getPrevIndex(current, mode, playedList) {
                if (mode === 'random') {
                    if (playedList.length >= songs.length) playedList.length = 0;
                    let available = songs.map((_,i)=>i).filter(i => !playedList.includes(i));
                    if (available.length === 0) { playedList.length = 0; available = songs.map((_,i)=>i); }
                    const rand = available[Math.floor(Math.random()*available.length)];
                    playedList.push(rand);
                    return rand;
                }
                if (mode === 'loop') return current;
                return (current - 1 + songs.length) % songs.length;
            }

            function startTest() { testAnswers = []; testCurrentQ = 0; showPage('test-page'); renderQuestion(); }
            window.startTest = startTest;

            function renderQuestion() {
                if (testCurrentQ >= testQuestions.length) { finishTest(); return; }
                const qData = testQuestions[testCurrentQ];
                document.getElementById('testProgress').textContent = '第 '+(testCurrentQ+1)+' / '+testQuestions.length+' 题';
                document.getElementById('testQuestion').textContent = qData.q;
                document.getElementById('testOptions').innerHTML = qData.opts.map((opt,i) => `<button class="test-option" onclick="selectAnswer(${i})">${String.fromCharCode(65+i)}. ${opt}</button>`).join('');
            }

            window.selectAnswer = function(optIndex) {
                const qData = testQuestions[testCurrentQ];
                const scores = qData.scores[optIndex];
                const dims = qData.dims;
                let scoreObj = {};
                if (Array.isArray(scores)) dims.forEach((d,i)=>{ scoreObj[d]=scores[i]; });
                else scoreObj[dims[0]] = scores;
                testAnswers.push(scoreObj);
                testCurrentQ++;
                renderQuestion();
            };

            function finishTest() {
                let totalScores = { energy:0, social:0, hope:0, romance:0, freedom:0 };
                let counts = { energy:0, social:0, hope:0, romance:0, freedom:0 };
                testAnswers.forEach(ans => { for(let d in ans) if(totalScores.hasOwnProperty(d)) { totalScores[d]+=ans[d]; counts[d]++; } });
                let userVector = {};
                for(let d in totalScores) userVector[d] = counts[d]>0 ? totalScores[d]/counts[d] : 3;
                let bestIndex = 0, bestDist = Infinity;
                let allDists = songs.map((s,i) => {
                    let dist = 0;
                    for(let d in userVector) dist += Math.pow(userVector[d]-s.tags[d],2);
                    return {index:i, dist:Math.sqrt(dist)};
                });
                allDists.forEach(d => { if(d.dist < bestDist) { bestDist=d.dist; bestIndex=d.index; } });
                let tied = allDists.filter(d => Math.abs(d.dist-bestDist)<0.001);
                if (tied.length>1) bestIndex = tied[Math.floor(Math.random()*tied.length)].index;
                resultPlayerIndex = bestIndex;
                resultPlayedRandom = [];
                document.getElementById('resSongName').textContent = songs[bestIndex].name;
                document.getElementById('resSongArtist').textContent = songs[bestIndex].artist.replace(/_/g,'/');
                document.getElementById('resMood').textContent = '心情匹配：'+songs[bestIndex].desc;
                const qrWrap = document.getElementById('qrCodeWrap');
                qrWrap.innerHTML = '';
                const qrURL = window.location.href.split('?')[0] + '?song='+encodeURIComponent(songs[bestIndex].name);
                new QRCode(qrWrap, { text:qrURL, width:140, height:140, colorDark:'#3a5a1f', colorLight:'#ffffff', correctLevel:QRCode.CorrectLevel.M });
                showPage('test-result-page');
                loadSong(resultPlayerIndex, resultAudioEl, 'discCoverResult', 'pSongNameResult', 'pSongArtistResult', 'progressBarResult', 'curTimeResult', 'totalTimeResult');
                playAudio(resultAudioEl, 'discOuterResult', 'btnPlayResult');
            }

            window.saveResultImage = function() {
                html2canvas(document.getElementById('resultCapture'), { backgroundColor:'#F9FEFF', scale:2, useCORS:true, allowTaint:true }).then(canvas => {
                    const link = document.createElement('a');
                    link.download = '18trip心情歌曲_'+songs[resultPlayerIndex].name+'.png';
                    link.href = canvas.toDataURL('image/png');
                    link.click();
                    showToast('图片已保存 (｡･ω･｡)');
                }).catch(()=> showToast('保存失败，请重试'));
            };

                renderSongList();
                setupMainPlayer();
                setupResultPlayer();
                const params = new URLSearchParams(window.location.search);
                const songParam = params.get('song');
                if (songParam) {
                    const foundIndex = songs.findIndex(s => s.name === songParam);
                    if (foundIndex >= 0) {
                        resultPlayerIndex = foundIndex;
                        showPage('test-result-page');
                        loadSong(foundIndex, resultAudioEl, 'discCoverResult', 'pSongNameResult', 'pSongArtistResult', 'progressBarResult', 'curTimeResult', 'totalTimeResult');
                        playAudio(resultAudioEl, 'discOuterResult', 'btnPlayResult');
                        return;
                    }
                }
                showPage('home-page');
            }
            init();
        })();
